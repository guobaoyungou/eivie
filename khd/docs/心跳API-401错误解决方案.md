# 心跳API 401错误 - 问题解决方案

> **问题描述**：前端调用心跳API时返回 401 错误："缺少设备Token"  
> **解决时间**：2026-02-02  
> **状态**：✅ 已解决

---

## 🔍 问题现象

### 前端日志显示

```javascript
心跳数据: { 
  device_token: 'f6a02fbf7682ec1eada81df66e9deeb7', 
  deviceId: '8C32231ED0E9' 
}
```

### 服务器返回

```json
{
  "code": 401,
  "msg": "缺少设备Token"
}
```

---

## ✅ 根本原因

**Token 传递位置错误！**

前端将 `device_token` 放在了**请求体**中，但后端要求必须放在**请求头**中。

---

## 🔧 解决方案

### ❌ 错误代码（修改前）

```javascript
// 错误：Token 放在请求体中
const heartbeatData = {
  device_token: 'f6a02fbf7682ec1eada81df66e9deeb7',  // ❌ 错误位置
  deviceId: '8C32231ED0E9',
  status: 'online'
};

const response = await axios.post(
  'http://192.168.11.222/api/ai_travel_photo/device/heartbeat',
  heartbeatData  // ❌ Token 在请求体中
);
```

### ✅ 正确代码（修改后）

```javascript
// 正确：Token 放在请求头中
const heartbeatData = {
  // device_token 不应该放在这里
  status: 'online',
  cpu_usage: 45.5,
  memory_usage: 68.2,
  timestamp: Math.floor(Date.now() / 1000)
};

const deviceToken = localStorage.getItem('device_token');  // 从存储中获取

const response = await axios.post(
  'http://192.168.11.222/api/ai_travel_photo/device/heartbeat',
  heartbeatData,
  {
    headers: {
      'Content-Type': 'application/json',
      'Device-Token': deviceToken  // ✅ 正确：放在请求头中
    }
  }
);
```

```javascript
/**
 * 发送设备心跳 - 正确实现
 */
async function sendHeartbeat() {
  try {
    // 1. 获取设备令牌
    const deviceToken = localStorage.getItem('device_token');
    
    if (!deviceToken) {
      console.error('设备未注册，请先注册设备');
      return false;
    }
    
    // 2. 构建心跳数据（不包含 token）
    const heartbeatData = {
      status: 'online',
      cpu_usage: getCpuUsage(),
      memory_usage: getMemoryUsage(),
      disk_usage: getDiskUsage(),
      network_status: 'good',
      upload_queue: getUploadQueueCount(),
      timestamp: Math.floor(Date.now() / 1000)
    };
    
    // 3. 发送请求 - Token 放在请求头
    const response = await axios.post(
      'http://192.168.11.222/api/ai_travel_photo/device/heartbeat',
      heartbeatData,
      {
        headers: {
          'Content-Type': 'application/json',
          'Device-Token': deviceToken  // ✅ 关键：Token 在请求头
        },
        timeout: 10000
      }
    );
    
    // 4. 处理响应
    if (response.data.code === 200) {
      console.log('心跳成功');
      return true;
    } else {
      console.error('心跳失败:', response.data.msg);
      return false;
    }
    
  } catch (error) {
    console.error('心跳异常:', error.message);
    
    // 处理 401 错误
    if (error.response && error.response.data.code === 401) {
      console.log('Token 失效，需要重新注册');
      localStorage.removeItem('device_token');
      // 触发重新注册
      await registerDevice();
    }
    
    return false;
  }
}
```

## 🎯 关键要点

### 1. Token 位置

| 位置 | 是否正确 | 说明 |
|------|---------|------|
| 请求头 `Device-Token` | ✅ 正确 | **必须使用这种方式** |
| 请求体 `device_token` | ❌ 错误 | 后端不会读取 |
| URL参数 `?token=xxx` | ❌ 错误 | 不安全，不推荐 |

### 2. 请求头名称

**大小写敏感！必须完全匹配：**

```javascript
✅ 正确：'Device-Token'
❌ 错误：'device_token'
❌ 错误：'device-token'
❌ 错误：'DEVICE-TOKEN'
❌ 错误：'DeviceToken'
```

### 3. Token 获取方式

```javascript
// ✅ 正确：从注册接口的响应中获取
const registerResponse = await axios.post('/api/ai_travel_photo/device/register', {
  device_code: 'DEVICE_001',
  device_id: 'xxx',
  bid: 1,
  aid: 1
});

if (registerResponse.data.code === 200) {
  const deviceToken = registerResponse.data.data.device_token;
  localStorage.setItem('device_token', deviceToken);  // 保存到本地
}

// ✅ 正确：后续从本地存储中读取
const deviceToken = localStorage.getItem('device_token');
```

## 💡 最佳实践

### 1. 封装心跳服务

```javascript
class HeartbeatService {
  constructor() {
    this.timer = null;
    this.interval = 60000; // 60秒
    this.isRunning = false;
  }
  
  start() {
    if (this.isRunning) {
      console.warn('心跳服务已在运行');
      return;
    }
    
    // 立即发送一次
    this.send();
    
    // 启动定时器
    this.timer = setInterval(() => {
      this.send();
    }, this.interval);
    
    this.isRunning = true;
    console.log('心跳服务已启动');
  }
  
  stop() {
    if (this.timer) {
      clearInterval(this.timer);
      this.timer = null;
    }
    this.isRunning = false;
    console.log('心跳服务已停止');
  }
  
  async send() {
    try {
      const deviceToken = localStorage.getItem('device_token');
      
      if (!deviceToken) {
        console.error('未找到设备令牌');
        return false;
      }
      
      const response = await axios.post(
        'http://192.168.11.222/api/ai_travel_photo/device/heartbeat',
        {
          status: 'online',
          timestamp: Math.floor(Date.now() / 1000)
        },
        {
          headers: {
            'Content-Type': 'application/json',
            'Device-Token': deviceToken
          },
          timeout: 10000
        }
      );
      
      if (response.data.code === 200) {
        console.log('[心跳] 成功');
        return true;
      } else if (response.data.code === 401) {
        console.error('[心跳] Token失效');
        this.stop();
        // 触发重新注册
        window.dispatchEvent(new Event('device-unauthorized'));
        return false;
      } else {
        console.error('[心跳] 失败:', response.data.msg);
        return false;
      }
      
    } catch (error) {
      console.error('[心跳] 异常:', error.message);
      return false;
    }
  }
}

// 使用示例
const heartbeat = new HeartbeatService();
heartbeat.start();

// 监听设备未授权事件
window.addEventListener('device-unauthorized', async () => {
  console.log('检测到设备未授权，重新注册...');
  await registerDevice();
  heartbeat.start();
});
```

### 2. 统一 API 客户端

```javascript
class ApiClient {
  constructor(baseURL) {
    this.client = axios.create({
      baseURL: baseURL,
      timeout: 10000,
      headers: {
        'Content-Type': 'application/json'
      }
    });
    
    // 请求拦截器：自动添加 Device-Token
    this.client.interceptors.request.use(config => {
      const deviceToken = localStorage.getItem('device_token');
      if (deviceToken) {
        config.headers['Device-Token'] = deviceToken;
      }
      return config;
    });
    
    // 响应拦截器：统一错误处理
    this.client.interceptors.response.use(
      response => response,
      error => {
        if (error.response?.data?.code === 401) {
          console.error('设备未授权');
          window.dispatchEvent(new Event('device-unauthorized'));
        }
        return Promise.reject(error);
      }
    );
  }
  
  // 心跳
  async heartbeat(data) {
    return await this.client.post('/api/ai_travel_photo/device/heartbeat', data);
  }
  
  // 获取配置
  async getConfig() {
    return await this.client.get('/api/ai_travel_photo/device/config');
  }
  
  // 上传文件
  async upload(formData) {
    return await this.client.post('/api/ai_travel_photo/device/upload', formData, {
      headers: { 'Content-Type': 'multipart/form-data' }
    });
  }
}

// 使用示例
const api = new ApiClient('http://192.168.11.222');

// 发送心跳（自动添加 Token）
const response = await api.heartbeat({
  status: 'online',
  timestamp: Math.floor(Date.now() / 1000)
});
```
