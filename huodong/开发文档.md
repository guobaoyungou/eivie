## 一、项目概述

### 1.1 项目简介

`huodong` 是一个现场活动大屏幕互动系统，支持签到、上墙、投票、抽奖、游戏、红包雨等多种互动功能。该系统采用 **PHP + MySQL + Vue.js (管理后台)** 技术栈，实现了 **多租户架构**，支持管理后台、大屏幕端、手机端的三端分离。

### 1.2 多租户架构

系统采用 **平台-商家-门店-用户** 四级多租户架构：

```
┌─────────────────────────────────────────────────────────────┐
│                      平台层 (Platform)                       │
│                         aid = 0/1                           │
│              管理所有商家(tenant)，提供基础服务                 │
└─────────────────────────────────────────────────────────────┘
                              │
          ┌───────────────────┼───────────────────┐
          ▼                   ▼                   ▼
┌─────────────────┐  ┌─────────────────┐  ┌─────────────────┐
│   商家 A        │  │   商家 B        │  │   商家 C        │
│   bid = 1       │  │   bid = 2       │  │   bid = 3       │
│  (Tenant A)     │  │  (Tenant B)     │  │  (Tenant C)     │
└─────────────────┘  └─────────────────┘  └─────────────────┘
          │                   │                   │
          ▼                   ▼                   ▼
┌─────────────────┐  ┌─────────────────┐  ┌─────────────────┐
│   门店 1-1      │  │   门店 2-1      │  │   门店 3-1      │
│   mdid = 1      │  │   mdid = ?      │  │   mdid = ?      │
│   活动 1,2,3    │  │   活动 ...      │  │   活动 ...      │
└─────────────────┘  └─────────────────┘  └─────────────────┘
          │                   │                   │
          ▼                   ▼                   ▼
┌─────────────────┐  ┌─────────────────┐  ┌─────────────────┐
│   参与者用户     │  │   参与者用户     │  │   参与者用户     │
│   openid        │  │   openid        │  │   openid        │
│   手机/签到      │  │   手机/签到      │  │   手机/签到      │
└─────────────────┘  └─────────────────┘  └─────────────────┘
```

### 1.3 核心标识符

| 标识符 | 说明 | 层级 | 示例 |
|--------|------|------|------|
| `aid` | 平台/管理员ID | 平台层 | 1 (平台管理员) |
| `bid` | 商家/租户ID | 商家层 | 1, 2, 3... |
| `mdid` | 门店ID | 门店层 | 1, 2, 3... |
| `activity_id` | 活动ID | 活动层 | 1, 2, 3... |
| `user_id` | 后台用户ID | 用户层 | 1, 2, 3... |
| `openid` | 微信用户标识 | 用户层 | oXXXX... |
| `access_code` | 活动访问码 | 活动层 | a1b2c3d4 |

### 1.4 数据隔离策略

```php
// 所有业务查询必须包含 aid + bid 隔离条件
$where = [
    ['aid', '=', $aid],   // 平台隔离
    ['bid', '=', $bid],   // 商家隔离
];
// 活动级数据还需加上
['activity_id', '=', $activity_id];
```

### 3.4 API请求封装 (api.js)

```javascript
// 基础配置
var BASE_URL = '/api/hd';
var TOKEN_KEY = 'hd_admin_token';

// Axios 实例配置
var http = axios.create({
    baseURL: BASE_URL,
    timeout: 30000,
    headers: { 'Content-Type': 'application/json' }
});

// 请求拦截器 - 自动注入 Token
http.interceptors.request.use(function(config) {
    var token = localStorage.getItem(TOKEN_KEY);
    if (token) {
        config.headers['Hd-Token'] = token;
    }
    return config;
});

// 响应拦截器 - 统一错误处理
http.interceptors.response.use(function(response) {
    var data = response.data;
    if (data && data.code !== undefined && data.code !== 0) {
        layui.layer.msg(data.msg || '操作失败', { icon: 2 });
        return Promise.reject(data);
    }
    return data;
}, function(error) {
    if (error.response) {
        var status = error.response.status;
        if (status === 401) {
            // Token 过期，跳转登录
        }
        if (status === 403) {
            // 无权限
        }
    }
    return Promise.reject(error);
});
```

public function getList(int $aid, int $bid, array $params = []): array

    $where = [

### 7.7 多租户安全要点

1. **强制隔离**: 所有数据操作必须包含 `aid` + `bid` 条件
2. **Token 不可伪造**: Token 包含加密的用户上下文
3. **权限验证**: 商家只能操作自己 `bid` 下的数据
4. **套餐限制**: 根据 `hd_business_config.plan_id` 限制资源使用
5. **活动访问码**: `access_code` 全局唯一，不可猜测

### 11.4 3D 渲染核心代码 (3d.js)

// 核心对象
var camera, scene, renderer;
var objects = [];  // 3D对象数组
var targets = {
    table: [],    // 表格布局
    torus: [],    // 隧道布局
    sphere: [],   // 球形布局
    helix: [],    // 螺旋布局
    grid: [],     // 网格布局
    cylinder: [], // 圆柱布局
    gene: []      // 基因布局
};

// 初始化
function init() {
    // 创建透视相机
    camera = new THREE.PerspectiveCamera(
        40, window.innerWidth / window.innerHeight, 1, 10000
    );
    camera.position.z = 3000;
    
    // 创建场景
    scene = new THREE.Scene();
    
    // 为每个头像创建3D对象
    for (var i = 0; i < table.length; i++) {
        var element = document.createElement('div');
        element.className = 'element';
        var img = document.createElement('img');
        img.src = table[i].src;
        element.appendChild(img);
        
        // 创建CSS3D对象
        var object = new THREE.CSS3DObject(element);
        object.position.x = Math.random() * 4000 - 2000;
        object.position.y = Math.random() * 4000 - 2000;
        object.position.z = Math.random() * 4000 - 2000;
        scene.add(object);
        objects.push(object);
    }
    
    // 创建渲染器
    renderer = new THREE.CSS3DRenderer();
    renderer.setSize(window.innerWidth, window.innerHeight);
    document.getElementById('container').appendChild(renderer.domElement);
}

// 切换造型动画
function transform(target, duration, style, interval) {
    TWEEN.removeAll();
    // 使用 TWEEN.js 实现平滑过渡动画
    for (var i = 0; i < objects.length; i++) {
        var object = objects[i];
        var target = targets[style][i];
        
        // 位置动画
        new TWEEN.Tween(object.position)
            .to({
                x: target.position.x,
                y: target.position.y,
                z: target.position.z
            }, Math.random() * duration + duration)
            .easing(TWEEN.Easing.Exponential.InOut)
            .start();
    }
}
